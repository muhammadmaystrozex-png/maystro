from kivy.app import App
from kivy.uix.boxlayout import BoxLayout
from kivy.uix.label import Label
from kivy.uix.textinput import TextInput
from kivy.uix.button import Button

# البيانات المعتمدة للمالك
OWNER_USER = "muhammad rahim"
OWNER_PASS = "maystro11"

class LegalApp(BoxLayout):
    def init(self, **kwargs):
        super().init(orientation='vertical', padding=20, spacing=10, **kwargs)
        
        # العنوان الرئيسي
        self.add_widget(Label(text="مساعدك القانوني", font_size=24))
        
        # حقول تسجيل الدخول
        self.user_input = TextInput(hint_text="اسم المستخدم", multiline=False)
        self.add_widget(self.user_input)
        
        self.pass_input = TextInput(hint_text="كلمة السر", password=True, multiline=False)
        self.add_widget(self.pass_input)
        
        # زر الدخول
        self.btn = Button(text="تسجيل دخول واستشارة AI", size_hint=(1, 0.5))
        self.btn.bind(on_press=self.check_login)
        self.add_widget(self.btn)
        
        # منطقة عرض النتائج
        self.result = Label(text="يرجى تسجيل الدخول للبدء")
        self.add_widget(self.result)

    def check_login(self, instance):
        # التحقق من البيانات المدخلة
        if self.user_input.text == OWNER_USER and self.pass_input.text == OWNER_PASS:
            self.result.text = "مرحباً بك يا محمد رحيم، جاري التحليل..."
        else:
            self.result.text = "بيانات الدخول خاطئة!"

class LegalAssistant(App):
    def build(self):
        return LegalApp()
